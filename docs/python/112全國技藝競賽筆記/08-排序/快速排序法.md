def quick_sort(lst):
    if len(lst) <= 1:
        return lst
    pivot = lst.pop(0)
    before = [i for i in lst if i < pivot]
    after = [i for i in lst if i >= pivot]
    return quick_sort(before) + [pivot] + quick_sort(after)


ary = [5, 3, 1, 2, 6, 4]

print("quick_sorted:")
print(*quick_sort(ary))
