еҪҲи·іиҰ–зӘ— (Modal) вҖӢ

рҹ“қ TL;DR

  • Modal = з–ҠеңЁй ҒйқўдёҠзҡ„з„Ұй»һе°Қи©ұжЎҶпјҢеӢҷеҝ…иҷ•зҗҶгҖҢй–Ӣе•ҹ/й—ңй–үгҖҒз„Ұй»һйҺ–е®ҡгҖҒиғҢжҷҜйҒ®зҪ©гҖҒEsc/й»һж“ҠиғҢжҷҜй—ңй–үгҖҚгҖӮ
  • еҲҘз”Ё display: none и—Ҹз„Ұй»һпјҡеҠ дёҠ aria-modal="true"гҖҒrole="dialog"пјҢдёҰеңЁй–Ӣе•ҹжҷӮжҠҠз„Ұй»һз§»еҲ°е°Қи©ұжЎҶгҖӮ
  • йҖІйҡҺпјҡеӨҡеұӨ ModalгҖҒеҸҜжҚІе…§е®№гҖҒйҳ»жӯўиғҢжҷҜжҚІеӢ•гҖҒеӢ•ж…ӢжҸ’е…Ҙе…§е®№гҖӮ

д»ҖйәјжҷӮеҖҷдҪҝз”Ё Modalпјҹ вҖӢ

  • зўәиӘҚйЎһиЎҢзӮәпјҡеҲӘйҷӨгҖҒйӣўй–ӢеүҚжҸҗзӨә
  • йӣҶдёӯжіЁж„ҸпјҡиЎЁе–®гҖҒзҷ»е…ҘгҖҒеҝ«йҖҹжӘўиҰ–е•Ҷе“Ғ
  • йқһйҳ»еЎһиіҮиЁҠдёҚи©Із”Ё ModalпјҡжҸҗзӨәеҸҜз”Ё Toast/Inline

<script>
const modal = document.getElementById('modal')
const openBtn = document.getElementById('open')
const closeBtn = document.getElementById('close')
const confirmBtn = document.getElementById('confirm')

function openModal() {
	modal.hidden = false
	document.body.style.overflow = 'hidden' // йҺ–е®ҡиғҢжҷҜж»ҫеӢ•
	modal.querySelector('button').focus()
}

function closeModal() {
	modal.hidden = true
	document.body.style.overflow = ''
	openBtn.focus()
}

openBtn.addEventListener('click', openModal)
closeBtn.addEventListener('click', closeModal)
confirmBtn.addEventListener('click', () => {
	alert('е·ІеҲӘйҷӨ')
	closeModal()
})

modal.addEventListener('click', (e) => {
	if (e.target === modal) closeModal() // й»һж“ҠиғҢжҷҜй—ңй–ү
})

document.addEventListener('keydown', (e) => {
	if (e.key === 'Escape' && !modal.hidden) closeModal()
})
</script>

  • й–Ӣе•ҹ/й—ңй–үпјҡжҢүйҲ•гҖҒEscгҖҒиғҢжҷҜй»һж“Ҡ
  • з„Ұй»һз®ЎзҗҶпјҡй–Ӣе•ҹжҷӮиҒҡз„ҰйҰ–еҖӢеҸҜдә’еӢ•е…ғзҙ пјӣй—ңй–үжҷӮйӮ„еҺҹеҲ°и§ёзҷјжәҗ
  • ж»ҫеӢ•жҺ§еҲ¶пјҡй–Ӣе•ҹжҷӮ body зҰҒжӯўж»ҫеӢ•пјҢй•·е…§е®№еҸҜеңЁе°Қи©ұжЎҶе…§ж»ҫ

еҜҰдҪңз„Ұй»һйҺ–е®ҡпјҡжҢү Tab жҷӮз„Ұй»һеңЁе°Қи©ұжЎҶе…§еҫӘз’°пјҢдёҚжңғи·‘еӣһиғҢжҷҜгҖӮ

рҹ’Ў еҸғиҖғзӯ”жЎҲиҲҮжҸҗзӨә

жҸҗзӨәпјҡ ж”¶йӣҶ dialog е…§еҸҜиҒҡз„Ұе…ғзҙ пјҢзӣЈиҒҪ keydown TabгҖӮ

javascript
const focusables = modal.querySelectorAll('button, [href], input, select, textarea')
modal.addEventListener('keydown', (e) => {
	if (e.key !== 'Tab') return
	const first = focusables[0]
	const last = focusables[focusables.length - 1]
	if (e.shiftKey && document.activeElement === first) { e.preventDefault(); last.focus() }
	if (!e.shiftKey && document.activeElement === last) { e.preventDefault(); first.focus() }
})
