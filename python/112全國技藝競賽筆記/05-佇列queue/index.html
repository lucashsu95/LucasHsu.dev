  1. е…ҲйҖІе…ҲеҮәпјҲFIFOпјүзү№жҖ§
  2. иҲүдҫӢпјҡ
    • жҺ’йҡҠиіјзҘЁпјҡ жғіеғҸдҪ еңЁйӣ»еҪұйҷўжҲ–зҒ«и»Ҡз«ҷжҺ’йҡҠиіјиІ·зҘЁгҖӮжҜҸеҖӢдәәдҫқж¬ЎжҺ’йҡҠпјҢз¬¬дёҖеҖӢжҺ’йҡҠзҡ„дәәжңҖе…ҲиІ·еҲ°зҘЁпјҢз„¶еҫҢдҫқеәҸеҗ‘еҫҢгҖӮйҖҷе°ұжҳҜдҪҮеҲ—зҡ„е…ҲйҖІе…ҲеҮәпјҲFIFOпјүзү№жҖ§пјҢеҚіе…ҲеҲ°е…Ҳеҫ—гҖӮ
    • йЈҹзү©й»һйӨҗпјҡ еңЁдёҖе®¶еҝ«йӨҗеә—иЈЎпјҢйЎ§е®ўеңЁж”¶йҠҖж©ҹеүҚжҺ’йҡҠй»һйӨҗгҖӮжҜҸеҖӢдәәйғҪжҢүз…§д»–еҖ‘еҲ°йҒ”зҡ„й ҶеәҸйҖІиЎҢй»һйӨҗпјҢиҖҢжңҚеӢҷе“Ўд№ҹжҢүз…§зӣёеҗҢзҡ„й ҶеәҸиҷ•зҗҶд»–еҖ‘зҡ„иЁӮе–®гҖӮйҖҷе°ұжҳҜдҪҮеҲ—зҡ„е…ҲйҖІе…ҲеҮәзү№жҖ§пјҢиЁӮе–®е°ҮжҢүз…§жҺ’йҡҠзҡ„й ҶеәҸдҫҶе®ҢжҲҗгҖӮ

import queue

е„Әе…Ҳж¬ҠдҪҮеҲ— вҖӢ

queue.PriorityQueue вҖӢ

  • гҖҗжј”з®—жі•зӯҶиЁҳ #4гҖ‘Priority Queue (е„Әе…Ҳж¬ҠдҪҮеҲ—) йҮҚй»һж•ҙзҗҶзӯҶиЁҳпјҢеҢ…еҗ«иҲҮеёёиҰӢзҡ„ Stack, Queue зҡ„жҜ”ијғ
python
import queue

# еүөе»әдёҖеҖӢе„Әе…Ҳж¬ҠдҪҮеҲ—
priority_queue = queue.PriorityQueue()

# е°Үе…ғзҙ ж”ҫе…ҘдҪҮеҲ—пјҢдёҰжҢҮе®ҡе„Әе…Ҳж¬Ҡ (е„Әе…Ҳж¬Ҡ,зү©е“Ғ)
priority_queue.put((3, "Item 3"))
priority_queue.put((1, "Item 1"))
priority_queue.put((2, "Item 2"))

# еҫһдҪҮеҲ—дёӯеҸ–еҮәе…ғзҙ 
item = priority_queue.get()
# еҫһдҪҮеҲ—дёӯеҸ–еҮә(1, "Item 1")пјҢеӣ зӮәе„Әе…Ҳж¬ҠжңҖдҪҺ
