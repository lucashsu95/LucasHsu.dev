1. еӢ•ж…ӢиҰҸеҠғж–№жі• вҖӢ

йҖҷжҳҜLISе•ҸйЎҢзҡ„жЁҷжә–и§Јжұәж–№жі•гҖӮйҖҡеёёдҪҝз”ЁдёҖеҖӢж•ёзө„дҫҶеӯҳе„Ід»ҘжҜҸеҖӢе…ғзҙ зөҗе°ҫзҡ„жңҖй•·йҒһеўһеӯҗеәҸеҲ—зҡ„й•·еәҰгҖӮ

python
def LIS_dp(nums):
    n = len(nums)
    dp = [1] * n
    for i in range(1, n):
        for j in range(i):
            if nums[i] > nums[j]:
                dp[i] = min(dp[i], dp[j] + 1)
    return max(dp)

def LIS_effiently(nums):
    dp = []
    for num in nums:
        i = 0
        j = len(dp)
        while i != j:
            mid = (i + j) // 2
            if dp[mid] < num:
                i = mid + 1
            else:
                j = mid

        if i == len(dp):
            dp.append(num)
        else:
            dp[i] = num

    return len(dp)

2. (жҺЁ)дәҢеҲҶжҹҘжүҫж–№жі• вҖӢ

йҖҷзЁ®ж–№жі•дҪҝз”ЁдәҢеҲҶжҹҘжүҫдҫҶе„ӘеҢ–еӢ•ж…ӢиҰҸеҠғи§Јжі•пјҢжёӣе°‘жҹҘжүҫе…ғзҙ зҡ„жҷӮй–“иҰҶйӣңеәҰгҖӮ

python
def LIS_with_bisect(nums):
    from bisect import bisect_left
    dp = []
    for num in nums:
        i = bisect_left(dp, num)
        if i == len(dp):
            dp.append(num)
        else:
            dp[i] = num
    return len(dp)

3. иІӘеҝғз®—жі•ж–№жі• вҖӢ

дҪҝз”ЁиІӘеҝғз®—жі•еҸҜд»ҘжүҫеҲ°LISзҡ„дёҖзЁ®йҖјиҝ‘ж–№жі•гҖӮе®ғзҡ„еҹәжң¬жҖқжғіжҳҜзӣЎйҮҸйҒёж“ҮжҜ”еүҚдёҖеҖӢе…ғзҙ еӨ§зҡ„е…ғзҙ пјҢж§Ӣе»әйҒһеўһеӯҗеәҸеҲ—гҖӮ

python
def LIS_greedy(arr):
    n = len(arr)
    lis = [arr[0]]

    for i in range(1, n):
        if arr[i] > lis[-1]:
            lis.append(arr[i])

    return len(lis)

4. еҲҶжІ»жі•ж–№жі• вҖӢ

еҲҶжІ»жі•д№ҹеҸҜд»Ҙз”ЁдҫҶи§ЈжұәLISе•ҸйЎҢпјҢе°Үе•ҸйЎҢеҲҶи§ЈзӮәжӣҙе°Ҹзҡ„еӯҗе•ҸйЎҢгҖӮ

python
def LIS_divide_conquer(arr):
    def _LIS(arr, n):
        if n == 1:
            return 1
        max_ending_here = 1
        for i in range(1, n):
            subproblem = _LIS(arr, i)
            if arr[i - 1] < arr[n - 1] and subproblem + 1 > max_ending_here:
                max_ending_here = subproblem + 1
        return max(max_ending_here, _LIS(arr, n - 1))

    n = len(arr)
    return _LIS(arr, n)

arr = [10, 22, 9, 33, 21, 50, 41, 60, 80]

print("LIS_with_bisect:", LIS_with_bisect(arr))
print("LIS_greedy:", LIS_greedy(arr))
print("LIS_divide_conquer:", LIS_divide_conquer(arr))
